4/// MF holding data structure
5#[derive(Debug, Clone, Serialize, Deserialize)]
6pub struct MFHolding {
7    /// Folio number
8    pub folio: String,
9
10    /// Fund name
11    pub fund: String,
12
13    /// Trading symbol
14    #[serde(rename = "tradingsymbol")]
15    pub trading_symbol: String,
16
17    /// Average price (average NAV at which units were purchased)
18    #[serde(rename = "average_price")]
19    pub average_price: f64,
20
21    /// Last price (current NAV)
22    #[serde(rename = "last_price")]
23    pub last_price: f64,
24
25    /// Last price date
26    #[serde(rename = "last_price_date")]
27    pub last_price_date: NaiveDate,
28
29    /// Quantity (units held)
30    pub quantity: f64,
31
32    /// P&L (profit and loss)
33    pub pnl: f64,
34

36    #[serde(rename = "pledged_quantity")]

47/// MF portfolio summary
48#[derive(Debug, Clone, Serialize, Deserialize)]
49pub struct MFPortfolioSummary {
50    /// Total investment value
51    pub total_investment: f64,
52
53    /// Total current value
54    pub total_current_value: f64,
55
56    /// Total P&L
57    pub total_pnl: f64,
58
59    /// Total P&L percentage
60    pub total_pnl_percentage: f64,
61
62    /// Number of holdings
63    pub holdings_count: usize,
64
65    /// Number of profitable holdings
66    pub profitable_holdings: usize,
67
68    /// Number of loss-making holdings
69    pub loss_holdings: usize,

72impl MFHolding {
73    /// Calculate current value of the holding
74    pub fn current_value(&self) -> f64 {
75        self.last_price * self.quantity
76    }
77
78    /// Calculate investment value (cost)
79    pub fn investment_value(&self) -> f64 {
80        self.average_price * self.quantity
81    }
82
83    /// Calculate P&L percentage
84    pub fn pnl_percentage(&self) -> f64 {
85        let investment = self.investment_value();
86        if investment > 0.0 {
87            (self.pnl / investment) * 100.0
88        } else {
89            0.0
90        }
91    }
92
93    /// Check if the holding is profitable
94    pub fn is_profitable(&self) -> bool {
95        self.pnl > 0.0
96    }
97
98    /// Check if the holding is in loss
99    pub fn is_loss(&self) -> bool {
100        self.pnl < 0.0
101    }
102
103    /// Get available quantity (non-pledged)
104    pub fn available_quantity(&self) -> f64 {
105        self.quantity - self.pledged_quantity
106    }
107
108    /// Check if any units are pledged
109    pub fn is_pledged(&self) -> bool {
110        self.pledged_quantity > 0.0
111    }
112
113    /// Get the day's change in NAV
114    pub fn nav_change(&self) -> f64 {
115        self.last_price - self.average_price
116    }
117
118    /// Get the day's change in NAV percentage
119    pub fn nav_change_percentage(&self) -> f64 {
120        if self.average_price > 0.0 {
121            ((self.last_price - self.average_price) / self.average_price) * 100.0
122        } else {
123            0.0
124        }
125    }
126
127    /// Calculate absolute P&L
128    pub fn abs_pnl(&self) -> f64 {
129        self.pnl.abs()
130    }
131
132    /// Get holding weight in portfolio (requires portfolio value)
133    pub fn weight_in_portfolio(&self, total_portfolio_value: f64) -> f64 {
134        if total_portfolio_value > 0.0 {
135            (self.current_value() / total_portfolio_value) * 100.0
136        } else {
137            0.0
138        }
139    }
140
141    /// Check if this is a small holding (less than specified value)
142    pub fn is_small_holding(&self, threshold: f64) -> bool {
143        self.current_value() < threshold
144    }
145
146    /// Calculate redemption value for specified units
147    pub fn redemption_value(&self, units: f64) -> f64 {
148        if units <= self.available_quantity() {
149            self.last_price * units
150        } else {
151            0.0 // Cannot redeem more than available
152        }
153    }
154}

156impl MFHoldings {
157    /// Calculate portfolio summary
158    pub fn portfolio_summary(&self) -> MFPortfolioSummary {
159        let total_investment = self.holdings.iter().map(|h| h.investment_value()).sum();
160        let total_current_value = self.holdings.iter().map(|h| h.current_value()).sum();
161        let total_pnl = self.holdings.iter().map(|h| h.pnl).sum();
162
163        let total_pnl_percentage = if total_investment > 0.0 {
164            (total_pnl / total_investment) * 100.0
165        } else {
166            0.0
167        };
168
169        let profitable_holdings = self.holdings.iter().filter(|h| h.is_profitable()).count();
170        let loss_holdings = self.holdings.iter().filter(|h| h.is_loss()).count();
171
172        MFPortfolioSummary {
173            total_investment,
174            total_current_value,
175            total_pnl,
176            total_pnl_percentage,
177            holdings_count: self.holdings.len(),
178            profitable_holdings,
179            loss_holdings,
180        }
181    }
182
183    /// Get profitable holdings
184    pub fn profitable_holdings(&self) -> Vec<&MFHolding> {
185        self.holdings.iter().filter(|h| h.is_profitable()).collect()
186    }
187
188    /// Get loss-making holdings
189    pub fn loss_holdings(&self) -> Vec<&MFHolding> {
190        self.holdings.iter().filter(|h| h.is_loss()).collect()
191    }
192
193    /// Get holdings by fund name pattern
194    pub fn holdings_by_fund_pattern(&self, pattern: &str) -> Vec<&MFHolding> {
195        self.holdings
196            .iter()
197            .filter(|h| h.fund.to_lowercase().contains(&pattern.to_lowercase()))
198            .collect()
199    }
200
201    /// Get top performers by P&L percentage
202    pub fn top_performers(&self, limit: usize) -> Vec<&MFHolding> {
203        let mut holdings = self.holdings.iter().collect::<Vec<_>>();
204        holdings.sort_by(|a, b| {
205            b.pnl_percentage()
206                .partial_cmp(&a.pnl_percentage())
207                .unwrap_or(std::cmp::Ordering::Equal)
208        });
209        holdings.into_iter().take(limit).collect()
210    }
211
212    /// Get worst performers by P&L percentage
213    pub fn worst_performers(&self, limit: usize) -> Vec<&MFHolding> {
214        let mut holdings = self.holdings.iter().collect::<Vec<_>>();
215        holdings.sort_by(|a, b| {
216            a.pnl_percentage()
217                .partial_cmp(&b.pnl_percentage())
218                .unwrap_or(std::cmp::Ordering::Equal)
219        });
220        holdings.into_iter().take(limit).collect()
221    }
222
223    /// Get largest holdings by current value
224    pub fn largest_holdings(&self, limit: usize) -> Vec<&MFHolding> {
225        let mut holdings = self.holdings.iter().collect::<Vec<_>>();
226        holdings.sort_by(|a, b| {
227            b.current_value()
228                .partial_cmp(&a.current_value())
229                .unwrap_or(std::cmp::Ordering::Equal)
230        });
231        holdings.into_iter().take(limit).collect()
232    }
233
234    /// Get small holdings below threshold
235    pub fn small_holdings(&self, threshold: f64) -> Vec<&MFHolding> {
236        self.holdings
237            .iter()
238            .filter(|h| h.is_small_holding(threshold))
239            .collect()
240    }
241
242    /// Find holding by folio number
243    pub fn find_by_folio(&self, folio: &str) -> Option<&MFHolding> {
244        self.holdings.iter().find(|h| h.folio == folio)
245    }
246
247    /// Find holding by trading symbol
248    pub fn find_by_symbol(&self, symbol: &str) -> Option<&MFHolding> {
249        self.holdings.iter().find(|h| h.trading_symbol == symbol)
250    }
251
252    /// Get total pledged value
253    pub fn total_pledged_value(&self) -> f64 {
254        self.holdings
255            .iter()
256            .map(|h| h.pledged_quantity * h.last_price)
257            .sum()
258    }
259
260    /// Get total available value (non-pledged)
261    pub fn total_available_value(&self) -> f64 {
262        self.holdings
263            .iter()
264            .map(|h| h.available_quantity() * h.last_price)
265            .sum()
266    }
