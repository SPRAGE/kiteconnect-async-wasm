39/// Historical data request parameters (v1.0.3 enhanced)
40///
41/// This struct provides a structured way to request historical OHLCV data with
42/// precise datetime control and optional parameters. It replaces the previous
43/// approach of passing multiple string parameters.
44///
45/// # Benefits over Legacy API
46///
47/// - **Type Safety**: Compile-time validation of parameters
48/// - **Precise Timing**: Hour/minute/second precision with `NaiveDateTime`
49/// - **Builder Pattern**: Fluent API for optional parameters
50/// - **Better IDE Support**: Autocomplete and documentation
51///
52/// # Date/Time Handling
53///
54/// The API expects datetime in IST (Indian Standard Time). When using `NaiveDateTime`,
55/// ensure your timestamps represent IST. The API will treat naive datetimes as IST.
56///
57/// # Instrument Tokens
58///
59/// Instrument tokens are unique identifiers for tradable instruments. You can obtain
60/// them from the instruments API or by searching instruments by symbol.
61///
62/// # Examples
63///
64/// ## Basic Daily Data Request
65///
66/// ```rust,no_run
67/// use kiteconnect_async_wasm::models::market_data::HistoricalDataRequest;
68/// use kiteconnect_async_wasm::models::common::Interval;
69/// use chrono::NaiveDateTime;
70///
71/// # fn example() -> Result<(), Box<dyn std::error::Error>> {
72/// let request = HistoricalDataRequest::new(
73///     738561,  // RELIANCE
74///     NaiveDateTime::parse_from_str("2023-11-01 09:15:00", "%Y-%m-%d %H:%M:%S")?,
75///     NaiveDateTime::parse_from_str("2023-11-30 15:30:00", "%Y-%m-%d %H:%M:%S")?,
76///     Interval::Day,
77/// );
78/// # Ok(())
79/// # }
80/// ```
81///
82/// ## Intraday Data with Options
83///
84/// ```rust,no_run
85/// use kiteconnect_async_wasm::models::market_data::HistoricalDataRequest;
86/// use kiteconnect_async_wasm::models::common::Interval;
87/// use chrono::NaiveDateTime;
88///
89/// # fn example() -> Result<(), Box<dyn std::error::Error>> {
90/// let request = HistoricalDataRequest::new(
91///     738561,
92///     NaiveDateTime::parse_from_str("2023-11-20 09:00:00", "%Y-%m-%d %H:%M:%S")?,
93///     NaiveDateTime::parse_from_str("2023-11-20 16:00:00", "%Y-%m-%d %H:%M:%S")?,
94///     Interval::FiveMinute,
95/// ).continuous(true)  // Include pre/post market
96///  .with_oi(false);   // No open interest for equity
97/// # Ok(())
98/// # }
99/// ```
100#[derive(Debug, Clone, Serialize, Deserialize)]
101pub struct HistoricalDataRequest {
102    /// Instrument token - unique identifier for the trading instrument
103    ///
104    /// This is a numeric identifier assigned to each tradable instrument.
105    /// You can obtain instrument tokens from:
106    /// - The instruments API (`client.instruments()`)
107    /// - Instrument search functionality
108    /// - KiteConnect documentation
109    ///
110    /// Example: 738561 for RELIANCE on NSE
111    #[serde(rename = "instrument_token")]
112    pub instrument_token: u32,
113
114    /// From date and time (IST)
115    ///
116    /// The start date and time for the historical data request.
117    /// Must be in Indian Standard Time (IST). For daily data,
118    /// you can use any time but 09:15:00 is recommended.
119    ///
120    /// # Limits
121    /// - Intraday data: Limited to recent days (varies by broker)
122    /// - Daily data: Several years of history available
123    /// - Cannot be in the future
124    pub from: NaiveDateTime,
125
126    /// To date and time (IST)
127    ///
128    /// The end date and time for the historical data request.
129    /// Must be in Indian Standard Time (IST) and after `from` date.
130    /// For daily data, you can use any time but 15:30:00 is recommended.
131    pub to: NaiveDateTime,
132
133    /// Time interval for the historical data
134    ///
135    /// Determines the granularity of the returned candles:
136    /// - `Minute`: 1-minute candles (for intraday analysis)
137    /// - `ThreeMinute`, `FiveMinute`, etc.: Various intraday intervals
138    /// - `Day`: Daily candles (for longer-term analysis)
139    ///
140    /// Note: Shorter intervals generate more data points and may hit
141    /// API rate limits faster.
142    pub interval: Interval,
143
144    /// Continuous data flag (for futures)
145    ///
146    /// When `true`, provides continuous contract data for futures instruments
147    /// by stitching together data from different contract expiries. This gives
148    /// a continuous price series without gaps from contract rollovers.
149    ///
150    /// - `true`: Continuous contract data (recommended for futures analysis)
151    /// - `false`: Individual contract data (default)
152    /// - `None`: Uses API default (false)
153    ///
154    /// Only applicable to futures instruments. Has no effect on equity/options.
155    pub continuous: Option<bool>,
156
157    /// Open interest data flag (for derivatives)
158    ///
159    /// When `true`, includes open interest data in the response for derivatives
160    /// instruments (futures and options). Open interest represents the total
161    /// number of outstanding contracts.
162    ///
163    /// - `true`: Include open interest data
164    /// - `false`: Exclude open interest data (smaller response)
165    /// - `None`: Uses API default (false)
166    ///
167    /// Only applicable to futures and options. Has no effect on equity instruments.
168    pub oi: Option<bool>,
169}

171/// Historical candle data point
172///
173/// Represents a single OHLCV (Open, High, Low, Close, Volume) data point
174/// for a specific time period. Each candle contains price and volume information
175/// for the specified interval.
176///
177/// # Data Quality
178///
179/// - Prices are adjusted for corporate actions (splits, bonuses, dividends)
180/// - Volume is in number of shares/contracts traded
181/// - Open interest (if available) is the total outstanding contracts
182///
183/// # Example
184///
185/// ```rust,no_run
186/// use kiteconnect_async_wasm::models::market_data::Candle;
187/// use chrono::{DateTime, Utc};
188///
189/// # fn example(candle: Candle) {
190/// println!("Date: {}", candle.date.format("%Y-%m-%d %H:%M:%S"));
191/// println!("OHLC: {}/{}/{}/{}", candle.open, candle.high, candle.low, candle.close);
192/// println!("Volume: {}", candle.volume);
193///
194/// if let Some(oi) = candle.oi {
195///     println!("Open Interest: {}", oi);
196/// }
197///
198/// // Calculate price change
199/// let change = candle.close - candle.open;
200/// let change_pct = (change / candle.open) * 100.0;
201/// println!("Change: ₹{:.2} ({:.2}%)", change, change_pct);
202/// # }
203/// ```
204#[derive(Debug, Clone, Serialize)]
205pub struct Candle {
206    /// Timestamp in UTC
207    ///
208    /// The date and time for this candle in UTC. For daily candles,
209    /// this typically represents the market close time. For intraday
210    /// candles, it represents the end time of the interval.
211    ///
212    /// Note: Even though stored as UTC, the original data is based on IST.
213    pub date: DateTime<Utc>,
214
215    /// Opening price for the interval
216    ///
217    /// The first traded price during this time interval.
218    /// For daily candles, this is the opening price of the trading day.
219    pub open: f64,
220
221    /// Highest price during the interval
222    ///
223    /// The maximum price reached during this time interval.
224    pub high: f64,
225
226    /// Lowest price during the interval
227    ///
228    /// The minimum price reached during this time interval.
229    pub low: f64,
230
231    /// Closing price for the interval
232    ///
233    /// The last traded price during this time interval.
234    /// For daily candles, this is the closing price of the trading day.
235    pub close: f64,
236
237    /// Volume traded during the interval
238    ///
239    /// Total number of shares/contracts traded during this time interval.
240    /// Higher volume often indicates stronger price moves and better liquidity.
241    pub volume: u64,
242
243    /// Open interest (for derivatives only)
244    ///
245    /// The total number of outstanding contracts at the end of this interval.
246    /// Only available for futures and options. `None` for equity instruments.
247    ///
248    /// Open interest helps gauge market sentiment:
249    /// - Increasing OI + Rising prices = Bullish sentiment
250    /// - Increasing OI + Falling prices = Bearish sentiment
251    /// - Decreasing OI = Position unwinding
252    pub oi: Option<u64>,
253}

255impl<'de> Deserialize<'de> for Candle {
256    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
257    where
258        D: Deserializer<'de>,
259    {
260        let value: Value = Deserialize::deserialize(deserializer)?;
261
262        // Handle array format: [date, open, high, low, close, volume] or [date, open, high, low, close, volume, oi]
263        if let Some(array) = value.as_array() {
264            if array.len() < 6 {
265                return Err(serde::de::Error::custom(format!(
266                    "Expected at least 6 elements in candle array, got {}",
267                    array.len()
268                )));
269            }
270
271            // Parse date (can be string or timestamp)
272            let date = if let Some(date_str) = array[0].as_str() {
273                // Try multiple date formats
274                DateTime::parse_from_rfc3339(date_str)
275                    .or_else(|_| {
276                        // Try parsing with +HHMM timezone format (like +0530)
277                        DateTime::parse_from_str(date_str, "%Y-%m-%dT%H:%M:%S%z")
278                    })
279                    .or_else(|_| {
280                        // Try parsing as ISO format with timezone
281                        DateTime::parse_from_str(date_str, "%Y-%m-%dT%H:%M:%S%:z")
282                    })
283                    .or_else(|_| {
284                        // Try parsing as simple date format and assume IST
285                        NaiveDateTime::parse_from_str(date_str, "%Y-%m-%d %H:%M:%S").map(|dt| {
286                            // Assume IST timezone (+05:30) and convert to UTC
287                            use chrono::FixedOffset;
288                            let ist = FixedOffset::east_opt(5 * 3600 + 30 * 60).unwrap();
289                            ist.from_local_datetime(&dt).unwrap().into()
290                        })
291                    })
292                    .map_err(|e| {
293                        serde::de::Error::custom(format!(
294                            "Failed to parse date '{}': {}",
295                            date_str, e
296                        ))
297                    })?
298                    .with_timezone(&Utc)
299            } else if let Some(timestamp) = array[0].as_i64() {
300                // Parse Unix timestamp
301                DateTime::from_timestamp(timestamp, 0).ok_or_else(|| {
302                    serde::de::Error::custom(format!("Invalid timestamp: {}", timestamp))
303                })?
304            } else {
305                return Err(serde::de::Error::custom("Date must be string or timestamp"));
306            };
307
308            let open = array[1]
309                .as_f64()
310                .ok_or_else(|| serde::de::Error::custom("Open price must be a number"))?;
311            let high = array[2]
312                .as_f64()
313                .ok_or_else(|| serde::de::Error::custom("High price must be a number"))?;
314            let low = array[3]
315                .as_f64()
316                .ok_or_else(|| serde::de::Error::custom("Low price must be a number"))?;
317            let close = array[4]
318                .as_f64()
319                .ok_or_else(|| serde::de::Error::custom("Close price must be a number"))?;
320            let volume = array[5]
321                .as_u64()
322                .ok_or_else(|| serde::de::Error::custom("Volume must be a positive integer"))?;
323
324            // Open interest is optional (7th element)
325            let oi = if array.len() > 6 {
326                array[6].as_u64()
327            } else {
328                None
329            };
330
331            Ok(Candle {
332                date,
333                open,
334                high,
335                low,
336                close,
337                volume,
338                oi,
339            })
340        } else {
341            // Handle object format for backwards compatibility
342            #[derive(Deserialize)]
343            struct CandleObject {
344                date: DateTime<Utc>,
345                open: f64,
346                high: f64,
347                low: f64,
348                close: f64,
349                volume: u64,
350                oi: Option<u64>,
351            }
352
353            let obj: CandleObject = serde_json::from_value(value).map_err(|e| {
354                serde::de::Error::custom(format!("Failed to deserialize candle object: {}", e))
355            })?;
356
357            Ok(Candle {
358                date: obj.date,
359                open: obj.open,
360                high: obj.high,
361                low: obj.low,
362                close: obj.close,
363                volume: obj.volume,
364                oi: obj.oi,
365            })
366        }
367    }
368}

427    /// Validate the date range against API limits for the specified interval
428    ///
429    /// Checks if the requested date range exceeds the maximum allowed days
430    /// for the specified interval according to KiteConnect API limits.
431    ///
432    /// # Returns
433    ///
434    /// - `Ok(())` if the date range is valid
435    /// - `Err(String)` with error description if validation fails
436    ///
437    /// # Example
438    ///
439    /// ```rust
440    /// use kiteconnect_async_wasm::models::market_data::HistoricalDataRequest;
441    /// use kiteconnect_async_wasm::models::common::Interval;
442    /// use chrono::NaiveDateTime;
443    ///
444    /// let request = HistoricalDataRequest::new(
445    ///     738561,
446    ///     NaiveDateTime::parse_from_str("2023-11-01 09:15:00", "%Y-%m-%d %H:%M:%S").unwrap(),
447    ///     NaiveDateTime::parse_from_str("2023-11-30 15:30:00", "%Y-%m-%d %H:%M:%S").unwrap(),
448    ///     Interval::Day,
449    /// );
450    ///
451    /// assert!(request.validate_date_range().is_ok());
452    /// ```
453    pub fn validate_date_range(&self) -> Result<(), String> {
454        if self.to <= self.from {
455            return Err("End date must be after start date".to_string());
456        }
457
458        if !self.interval.is_date_range_valid(&self.from, &self.to) {
459            let duration = self.to - self.from;
460            let days = duration.num_days();
461            let max_days = self.interval.max_days_allowed();
462
463            return Err(format!(
464                "Date range of {} days exceeds maximum allowed {} days for {} interval",
465                days, max_days, self.interval
466            ));
467        }
468
469        Ok(())
470    }

515    /// Split a large date range into multiple smaller requests that respect API limits
516    ///
517    /// If the requested date range exceeds the maximum allowed for the interval,
518    /// this method splits it into multiple smaller requests that can be made separately.
519    ///
520    /// # Returns
521    ///
522    /// Vector of `HistoricalDataRequest` objects, each within API limits
523    ///
524    /// # Example
525    ///
526    /// ```rust
527    /// use kiteconnect_async_wasm::models::market_data::HistoricalDataRequest;
528    /// use kiteconnect_async_wasm::models::common::Interval;
529    /// use chrono::NaiveDateTime;
530    ///
531    /// // Request 200 days of 5-minute data (exceeds 90-day limit)
532    /// let request = HistoricalDataRequest::new(
533    ///     738561,
534    ///     NaiveDateTime::parse_from_str("2023-01-01 09:15:00", "%Y-%m-%d %H:%M:%S").unwrap(),
535    ///     NaiveDateTime::parse_from_str("2023-07-20 15:30:00", "%Y-%m-%d %H:%M:%S").unwrap(),
536    ///     Interval::FiveMinute,
537    /// );
538    ///
539    /// let sub_requests = request.split_into_valid_requests();
540    /// println!("Split into {} requests", sub_requests.len());
541    /// ```
542    pub fn split_into_valid_requests(&self) -> Vec<Self> {
543        let max_days = self.interval.max_days_allowed() as i64;
544        let total_duration = self.to - self.from;
545        let total_days = total_duration.num_days();
546
547        if total_days <= max_days {
548            // Already within limits
549            return vec![self.clone()];
550        }
551
552        let mut requests = Vec::new();
553        let mut current_from = self.from;
554
555        while current_from < self.to {
556            let max_to = current_from + chrono::Duration::days(max_days);
557            let current_to = if max_to > self.to { self.to } else { max_to };
558
559            let request = Self {
560                instrument_token: self.instrument_token,
561                from: current_from,
562                to: current_to,
563                interval: self.interval,
564                continuous: self.continuous,
565                oi: self.oi,
566            };
567
568            requests.push(request);
569
570            // Fix data duplication: Move to next day to avoid both-inclusive overlap
571            // For intraday intervals, move by appropriate time increment
572            current_from = match self.interval {
573                crate::models::common::Interval::Minute => {
574                    current_to + chrono::Duration::minutes(1)
575                }
576                crate::models::common::Interval::ThreeMinute => {
577                    current_to + chrono::Duration::minutes(3)
578                }
579                crate::models::common::Interval::FiveMinute => {
580                    current_to + chrono::Duration::minutes(5)
581                }
582                crate::models::common::Interval::TenMinute => {
583                    current_to + chrono::Duration::minutes(10)
584                }
585                crate::models::common::Interval::FifteenMinute => {
586                    current_to + chrono::Duration::minutes(15)
587                }
588                crate::models::common::Interval::ThirtyMinute => {
589                    current_to + chrono::Duration::minutes(30)
590                }
591                crate::models::common::Interval::SixtyMinute => {
592                    current_to + chrono::Duration::hours(1)
593                }
594                crate::models::common::Interval::Day => current_to + chrono::Duration::days(1),
595            };
596        }
597
598        requests
599    }

601    /// Split a large date range into multiple smaller requests in reverse chronological order
602    ///
603    /// This method splits the request from newest to oldest dates, which is more efficient
604    /// for early termination when data is not available for older periods.
605    ///
606    /// # Returns
607    ///
608    /// Vector of `HistoricalDataRequest` objects in reverse chronological order (newest first)
609    ///
610    /// # Example
611    ///
612    /// ```rust
613    /// use kiteconnect_async_wasm::models::market_data::HistoricalDataRequest;
614    /// use kiteconnect_async_wasm::models::common::Interval;
615    /// use chrono::NaiveDateTime;
616    ///
617    /// // Request 200 days of 5-minute data (exceeds 90-day limit)
618    /// let request = HistoricalDataRequest::new(
619    ///     738561,
620    ///     NaiveDateTime::parse_from_str("2023-01-01 09:15:00", "%Y-%m-%d %H:%M:%S").unwrap(),
621    ///     NaiveDateTime::parse_from_str("2023-07-20 15:30:00", "%Y-%m-%d %H:%M:%S").unwrap(),
622    ///     Interval::FiveMinute,
623    /// );
624    ///
625    /// let sub_requests = request.split_into_valid_requests_reverse();
626    /// println!("Split into {} requests (newest first)", sub_requests.len());
627    /// ```
628    pub fn split_into_valid_requests_reverse(&self) -> Vec<Self> {
629        let max_days = self.interval.max_days_allowed() as i64;
630        let total_duration = self.to - self.from;
631        let total_days = total_duration.num_days();
632
633        if total_days <= max_days {
634            // Already within limits
635            return vec![self.clone()];
636        }
637
638        let mut requests = Vec::new();
639        let mut current_to = self.to;
640
641        while current_to > self.from {
642            let min_from = current_to - chrono::Duration::days(max_days);
643            let current_from = if min_from < self.from {
644                self.from
645            } else {
646                min_from
647            };
648
649            let request = Self {
650                instrument_token: self.instrument_token,
651                from: current_from,
652                to: current_to,
653                interval: self.interval,
654                continuous: self.continuous,
655                oi: self.oi,
656            };
657
658            requests.push(request);
659
660            // Fix data duplication: Move to previous time increment to avoid overlap
661            current_to = match self.interval {
662                crate::models::common::Interval::Minute => {
663                    current_from - chrono::Duration::minutes(1)
664                }
665                crate::models::common::Interval::ThreeMinute => {
666                    current_from - chrono::Duration::minutes(3)
667                }
668                crate::models::common::Interval::FiveMinute => {
669                    current_from - chrono::Duration::minutes(5)
670                }
671                crate::models::common::Interval::TenMinute => {
672                    current_from - chrono::Duration::minutes(10)
673                }
674                crate::models::common::Interval::FifteenMinute => {
675                    current_from - chrono::Duration::minutes(15)
676                }
677                crate::models::common::Interval::ThirtyMinute => {
678                    current_from - chrono::Duration::minutes(30)
679                }
680                crate::models::common::Interval::SixtyMinute => {
681                    current_from - chrono::Duration::hours(1)
682                }
683                crate::models::common::Interval::Day => current_from - chrono::Duration::days(1),
684            };
685        }
686
687        requests
688    }

725    /// Create a helper for fetching data with automatic chunking
726    ///
727    /// This method creates a convenient interface for fetching historical data
728    /// that automatically handles large date ranges by splitting them into
729    /// multiple API calls. This provides the best user experience.
730    ///
731    /// # Arguments
732    ///
733    /// * `client` - The KiteConnect client to use for API calls
734    /// * `continue_on_error` - Whether to continue if a chunk fails (default: false)
735    ///
736    /// # Returns
737    ///
738    /// A `Result<HistoricalData>` containing all candles from the entire date range
739    ///
740    /// # Example
741    ///
742    /// ```rust,no_run
743    /// use kiteconnect_async_wasm::connect::KiteConnect;
744    /// use kiteconnect_async_wasm::models::market_data::HistoricalDataRequest;
745    /// use kiteconnect_async_wasm::models::common::Interval;
746    /// use chrono::NaiveDateTime;
747    ///
748    /// # #[tokio::main]
749    /// # async fn main() -> Result<(), Box<dyn std::error::Error>> {
750    /// let client = KiteConnect::new("api_key", "access_token");
751    ///
752    /// // Create request for large date range
753    /// let request = HistoricalDataRequest::new(
754    ///     738561,
755    ///     NaiveDateTime::parse_from_str("2023-01-01 09:15:00", "%Y-%m-%d %H:%M:%S")?,
756    ///     NaiveDateTime::parse_from_str("2023-12-31 15:30:00", "%Y-%m-%d %H:%M:%S")?,
757    ///     Interval::Day,
758    /// );
759    ///
760    /// // Fetch with automatic chunking - single call handles everything!
761    /// let all_data = request.fetch_with_chunking(&client, false).await?;
762    /// println!("Retrieved {} candles for the entire year!", all_data.candles.len());
763    /// # Ok(())
764    /// # }
765    /// ```
766    ///
767    /// # Features
768    ///
769    /// - **Transparent**: Works exactly like a single API call from user perspective
770    /// - **Automatic**: Detects when chunking is needed and handles it automatically
771    /// - **Fast Path**: Uses regular API for requests within limits (no overhead)
772    /// - **Robust**: Optional error recovery for failed chunks
773    /// - **Progress**: Logs progress for large requests (when debug enabled)
774    pub async fn fetch_with_chunking(
775        self,
776        client: &crate::connect::KiteConnect,
777        continue_on_error: bool,
778    ) -> crate::models::common::KiteResult<HistoricalData> {
779        client
780            .historical_data_chunked(self, continue_on_error)
781            .await
782    }
783
784    /// Convenience method for fetching data with default error handling
785    ///
786    /// This is equivalent to `fetch_with_chunking(client, false)` - stops on first error.
787    /// Use this for most cases where you want all data or nothing.
788    ///
789    /// # Example
790    ///
791    /// ```rust,no_run
792    /// use kiteconnect_async_wasm::connect::KiteConnect;
793    /// use kiteconnect_async_wasm::models::market_data::HistoricalDataRequest;
794    /// use kiteconnect_async_wasm::models::common::Interval;
795    /// use chrono::NaiveDateTime;
796    ///
797    /// # #[tokio::main]
798    /// # async fn main() -> Result<(), Box<dyn std::error::Error>> {
799    /// let client = KiteConnect::new("api_key", "access_token");
800    ///
801    /// let data = HistoricalDataRequest::new(
802    ///     738561,
803    ///     NaiveDateTime::parse_from_str("2023-01-01 09:15:00", "%Y-%m-%d %H:%M:%S")?,
804    ///     NaiveDateTime::parse_from_str("2024-01-01 15:30:00", "%Y-%m-%d %H:%M:%S")?,
805    ///     Interval::Day,
806    /// ).fetch(&client).await?;
807    ///
808    /// println!("Got {} candles", data.candles.len());
809    /// # Ok(())
810    /// # }
811    /// ```
812    pub async fn fetch(
813        self,
814        client: &crate::connect::KiteConnect,
815    ) -> crate::models::common::KiteResult<HistoricalData> {
816        self.fetch_with_chunking(client, false).await
817    }
