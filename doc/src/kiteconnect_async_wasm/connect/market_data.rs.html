1//! # Market Data Module
2//!
3//! This module provides comprehensive market data access for the KiteConnect API v1.0.3,
4//! offering both real-time and historical market information with full cross-platform support.
5//!
6//! ## Overview
7//!
8//! The market data module is the core component for accessing all market-related information
9//! including instruments, quotes, historical data, and market depth. It provides both legacy
10//! JSON-based APIs and modern strongly-typed APIs for enhanced developer experience.
11//!
12//! ## Key Features
13//!
14//! ### üîÑ **Dual API Support**
15//! - **Legacy API**: Returns `JsonValue` for backward compatibility
16//! - **Typed API**: Returns structured types with compile-time safety (methods ending in `_typed`)
17//!
18//! ### üìä **Comprehensive Data Coverage**
19//! - **Real-time Quotes**: Live prices, OHLC, volume, and market depth
20//! - **Historical Data**: OHLCV candlestick data with v1.0.3 enhanced API
21//! - **Instruments Master**: Complete instrument list with metadata
22//! - **Market Status**: Exchange timings and market state information
23//!
24//! ### üåê **Cross-Platform Optimization**
25//! - **Native**: Uses `csv` crate for efficient parsing with structured JSON output
26//! - **WASM**: Uses `csv-core` for no-std CSV parsing in browser environments
27//! - **Rate Limiting**: Automatic rate limiting respecting KiteConnect API limits
28//! - **Error Handling**: Comprehensive error types with context
29//!
30//! ## Platform-Specific Behavior
31//!
32//! ### Native Platform (tokio)
33//! ```rust,no_run
34//! # use kiteconnect_async_wasm::connect::KiteConnect;
35//! # #[tokio::main]
36//! # async fn main() -> Result<(), Box<dyn std::error::Error>> {
37//! # let client = KiteConnect::new("api_key", "access_token");
38//! // Instruments data is parsed server-side and returned as structured JSON
39//! let instruments = client.instruments(None).await?;
40//! // Returns JsonValue with array of instrument objects
41//! # Ok(())
42//! # }
43//! ```
44//!
45//! ### WASM Platform
46//! ```rust,no_run
47//! # use kiteconnect_async_wasm::connect::KiteConnect;
48//! # #[tokio::main]
49//! # async fn main() -> Result<(), Box<dyn std::error::Error>> {
50//! # let client = KiteConnect::new("api_key", "access_token");
51//! // CSV data is parsed client-side using csv-core for browser compatibility
52//! let instruments = client.instruments(None).await?;
53//! // Returns JsonValue with array of instrument objects (parsed from CSV)
54//! # Ok(())
55//! # }
56//! ```
57//!
58//! ## Available Methods
59//!
60//! ### Instruments and Market Data
61//! - [`instruments()`](KiteConnect::instruments) - Get complete instruments list (cached for performance)
62//! - [`mf_instruments()`](KiteConnect::mf_instruments) - Get mutual fund instruments
63//! - [`quote()`](KiteConnect::quote) / [`quote_typed()`](KiteConnect::quote_typed) - Real-time quotes
64//! - [`ohlc()`](KiteConnect::ohlc) / [`ohlc_typed()`](KiteConnect::ohlc_typed) - OHLC data
65//! - [`ltp()`](KiteConnect::ltp) / [`ltp_typed()`](KiteConnect::ltp_typed) - Last traded price
66//!
67//! ### Historical Data (Enhanced in v1.0.3)
68//! - [`historical_data()`](KiteConnect::historical_data) - Legacy historical data API
69//! - [`historical_data_typed()`](KiteConnect::historical_data_typed) - New structured request API
70//!
71//! ### Market Information
72//! - [`trigger_range()`](KiteConnect::trigger_range) - Get trigger range for instruments
73//! - [`instruments_margins()`](KiteConnect::instruments_margins) - Get margin requirements

278    /// Get the trigger range for a list of instruments
279    pub async fn trigger_range(
280        &self,
281        transaction_type: &str,
282        instruments: Vec<&str>,
283    ) -> Result<JsonValue> {
284        let mut params: Vec<(&str, &str)> = Vec::new();
285        params.push(("transaction_type", transaction_type));
286
287        for instrument in instruments {
288            params.push(("instruments", instrument));
289        }
290
291        let resp = self
292            .send_request_with_rate_limiting_and_retry(
293                KiteEndpoint::TriggerRange,
294                &[],
295                Some(params),
296                None,
297            )
298            .await
299            .map_err(|e| anyhow::anyhow!("Get trigger range failed: {:?}", e))?;
300        self.raise_or_return_json(resp).await
301    }

303    /// Get instruments list
304    #[cfg(all(feature = "native", not(target_arch = "wasm32")))]
305    pub async fn instruments(&self, exchange: Option<&str>) -> Result<JsonValue> {
306        // Check cache first if enabled
307        if let Some(ref cache_config) = self.cache_config {
308            if cache_config.enable_instruments_cache && exchange.is_none() {
309                // Only cache the full instruments list, not exchange-specific ones
310                if let Ok(cache_guard) = self.response_cache.lock() {
311                    if let Some(ref cache) = *cache_guard {
312                        if let Some(cached_data) = cache.get_instruments() {
313                            return Ok(cached_data);
314                        }
315                    }
316                }
317            }
318        }
319
320        let endpoint = KiteEndpoint::Instruments;
321
322        let path_segments = if let Some(exchange) = exchange {
323            vec![exchange]
324        } else {
325            vec![]
326        };
327
328        let resp = self
329            .send_request_with_rate_limiting_and_retry(endpoint, &path_segments, None, None)
330            .await
331            .map_err(|e| anyhow::anyhow!("Get instruments failed: {:?}", e))?;
332        let body = resp.text().await?;
333
334        // Parse CSV response
335        let mut rdr = ReaderBuilder::new().from_reader(body.as_bytes());
336        let mut result = Vec::new();
337
338        let headers = rdr.headers()?.clone();
339        for record in rdr.records() {
340            let record = record?;
341            let mut obj = serde_json::Map::new();
342
343            for (i, field) in record.iter().enumerate() {
344                if let Some(header) = headers.get(i) {
345                    obj.insert(header.to_string(), JsonValue::String(field.to_string()));
346                }
347            }
348            result.push(JsonValue::Object(obj));
349        }
350
351        let result_json = JsonValue::Array(result);
352
353        // Cache the result if enabled and it's the full instruments list
354        if let Some(ref cache_config) = self.cache_config {
355            if cache_config.enable_instruments_cache && exchange.is_none() {
356                if let Ok(mut cache_guard) = self.response_cache.lock() {
357                    if let Some(ref mut cache) = *cache_guard {
358                        cache.set_instruments(result_json.clone());
359                    }
360                }
361            }
362        }
363
364        Ok(result_json)
365    }

479    /// Retrieves historical candlestick data for an instrument
480    ///
481    /// Returns historical OHLCV (Open, High, Low, Close, Volume) data for a given
482    /// instrument within the specified date range and interval. This is useful for
483    /// backtesting, analysis, and charting applications.
484    ///
485    /// # Arguments
486    ///
487    /// * `instrument_token` - The instrument token (numeric ID) of the instrument
488    /// * `from_date` - Start date in YYYY-MM-DD format
489    /// * `to_date` - End date in YYYY-MM-DD format  
490    /// * `interval` - Time interval for candlesticks ("minute", "day", "3minute", "5minute", "10minute", "15minute", "30minute", "60minute")
491    /// * `continuous` - Whether to include pre-market and post-market data ("1" for true, "0" for false)
492    ///
493    /// # Returns
494    ///
495    /// A `Result<JsonValue>` containing historical data with fields like:
496    /// - `data` - Array of candlestick data points
497    ///   - `date` - ISO datetime string
498    ///   - `open` - Opening price
499    ///   - `high` - Highest price
500    ///   - `low` - Lowest price
501    ///   - `close` - Closing price
502    ///   - `volume` - Trading volume
503    ///
504    /// # Errors
505    ///
506    /// Returns an error if:
507    /// - The instrument token is invalid
508    /// - The date range is invalid or too large
509    /// - The interval is not supported
510    /// - Network request fails
511    /// - User is not authenticated
512    ///
513    /// # Example
514    ///
515    /// ```rust,no_run
516    /// use kiteconnect_async_wasm::connect::KiteConnect;
517    ///
518    /// # #[tokio::main]
519    /// # async fn main() -> Result<(), Box<dyn std::error::Error>> {
520    /// let client = KiteConnect::new("api_key", "access_token");
521    ///
522    /// // Get daily data for RELIANCE for the last month
523    /// let historical_data = client.historical_data(
524    ///     "738561",           // RELIANCE instrument token
525    ///     "2023-11-01",       // From date
526    ///     "2023-11-30",       // To date
527    ///     "day",              // Daily interval
528    ///     "0"                 // No continuous data
529    /// ).await?;
530    ///
531    /// println!("Historical data: {:?}", historical_data);
532    ///
533    /// // Access candlestick data
534    /// if let Some(data) = historical_data["data"].as_array() {
535    ///     for candle in data {
536    ///         println!("Date: {}, Close: {}, Volume: {}",
537    ///             candle["date"], candle["close"], candle["volume"]);
538    ///     }
539    /// }
540    /// # Ok(())
541    /// # }
542    /// ```
543    ///
544    /// # Notes
545    ///
546    /// - Historical data is subject to availability and may have limitations based on your subscription
547    /// - Large date ranges may be split into multiple requests automatically
548    /// - Intraday data older than a certain period may not be available
549    /// - Weekend and holiday data will not be included in the response

551        &self,
552        instrument_token: &str,
553        from_date: &str,
554        to_date: &str,
555        interval: &str,

561            ("continuous", continuous),
562        ];
563
564        let resp = self
565            .send_request_with_rate_limiting_and_retry(
566                KiteEndpoint::HistoricalData,
567                &[instrument_token, interval],
568                Some(params),
569                None,
570            )
571            .await
572            .map_err(|e| anyhow::anyhow!("Get historical data failed: {:?}", e))?;
573
574        self.raise_or_return_json(resp).await
575    }

582    /// # Arguments

591    /// - `market_depth` - Bid/Ask data with quantities

594    /// # Example
595    ///
596    /// ```rust,no_run
597    /// use kiteconnect_async_wasm::connect::KiteConnect;
598    ///
599    /// # #[tokio::main]
600    /// # async fn main() -> Result<(), Box<dyn std::error::Error>> {

606    /// println!("Quotes: {:?}", quotes);
607    /// # Ok(())
608    /// # }
609    /// ```
610    pub async fn quote(&self, instruments: Vec<&str>) -> Result<JsonValue> {
611        let params: Vec<_> = instruments.into_iter().map(|i| ("i", i)).collect();
612        let resp = self

871    /// Get historical data with typed response
872    ///
873    /// Returns strongly typed historical data instead of JsonValue.
874    ///
875    /// # Arguments
876    ///
877    /// * `request` - A `HistoricalDataRequest` containing all the parameters for the request
878    ///
879    /// # Returns
880    ///
881    /// A `KiteResult<HistoricalData>` containing typed historical data
882    ///
883    /// # Example
884    ///
885    /// ```rust,no_run
886    /// use kiteconnect_async_wasm::connect::KiteConnect;
887    /// use kiteconnect_async_wasm::models::market_data::HistoricalDataRequest;
888    /// use kiteconnect_async_wasm::models::common::Interval;
889    /// use chrono::NaiveDateTime;
890    ///
891    /// # #[tokio::main]
892    /// # async fn main() -> Result<(), Box<dyn std::error::Error>> {
893    /// let client = KiteConnect::new("api_key", "access_token");
894    ///
895    /// let request = HistoricalDataRequest::new(
896    ///     738561,             // RELIANCE instrument token
897    ///     NaiveDateTime::parse_from_str("2023-11-01 00:00:00", "%Y-%m-%d %H:%M:%S")?,
898    ///     NaiveDateTime::parse_from_str("2023-11-30 23:59:59", "%Y-%m-%d %H:%M:%S")?,
899    ///     Interval::Day,
900    /// ).continuous(false);
901    ///
902    /// let historical_data = client.historical_data_typed(request).await?;
903    ///
904    /// for candle in &historical_data.candles {
905    ///     println!("Date: {}, Close: {}, Volume: {}",
906    ///         candle.date, candle.close, candle.volume);
907    /// }
908    /// # Ok(())
909    /// # }
910    /// ```
911    pub async fn historical_data_typed(
912        &self,
913        request: HistoricalDataRequest,
914    ) -> KiteResult<HistoricalData> {
915        // Validate date range against API limits
916        if let Err(validation_error) = request.validate_date_range() {
917            return Err(crate::models::common::KiteError::input_exception(validation_error));
918        }
919
920        let mut params = Vec::new();
921        params.push(("from", request.from.format("%Y-%m-%d %H:%M:%S").to_string()));
922        params.push(("to", request.to.format("%Y-%m-%d %H:%M:%S").to_string()));
923
924        if let Some(continuous) = request.continuous {
925            params.push(("continuous", if continuous { "1" } else { "0" }.to_string()));
926        }
927
928        if let Some(oi) = request.oi {
929            params.push(("oi", if oi { "1" } else { "0" }.to_string()));
930        }
931
932        // Convert params to the expected format
933        let params_str: Vec<(&str, &str)> = params.iter().map(|(k, v)| (*k, v.as_str())).collect();
934
935        let resp = self
936            .send_request_with_rate_limiting_and_retry(
937                KiteEndpoint::HistoricalData,
938                &[
939                    &request.instrument_token.to_string(),
940                    &request.interval.to_string(),
941                ],
942                Some(params_str),
943                None,
944            )
945            .await?;
946
947        let json_response = self.raise_or_return_json_typed(resp).await?;
948
949        // Extract the data field from response
950        let data = json_response["data"].clone();
951        
952        // Parse the candles array directly
953        let candles: Vec<crate::models::market_data::Candle> = if data["candles"].is_array() {
954            // If data has a "candles" field
955            serde_json::from_value(data["candles"].clone()).map_err(KiteError::Json)?
956        } else if data.is_array() {
957            // If data is directly an array of candles
958            serde_json::from_value(data).map_err(KiteError::Json)?
959        } else {
960            return Err(KiteError::general("Invalid historical data format".to_string()));
961        };
962
963        // Create metadata from request parameters
964        let metadata = crate::models::market_data::HistoricalMetadata {
965            instrument_token: request.instrument_token,
966            symbol: format!("Token-{}", request.instrument_token), // We don't have the actual symbol
967            interval: request.interval,
968            count: candles.len(),
969        };
970
971        Ok(crate::models::market_data::HistoricalData {
972            candles,
973            metadata,
974        })
975    }

977    /// Retrieve historical data with automatic chunking for large date ranges
978    ///
979    /// This method automatically handles date ranges that exceed API limits by splitting them
980    /// into smaller chunks and making multiple API calls. Features intelligent optimizations
981    /// including reverse chronological processing and early termination for maximum efficiency.
982    ///
983    /// # Features
984    ///
985    /// - **Automatic Chunking**: Splits large requests into API-compliant chunks
986    /// - **Reverse Chronological**: Processes newest data first for early termination efficiency
987    /// - **Smart Early Termination**: Stops when empty chunks indicate no more data exists
988    /// - **Data Deduplication**: Prevents overlapping date ranges in both-inclusive API
989    /// - **Progress Logging**: Logs progress for large requests (when debug feature enabled)
990    /// - **Rate Limiting**: Respects rate limits between chunk requests
991    /// - **Error Handling**: Continues with remaining chunks if one fails (configurable)
992    ///
993    /// # Arguments
994    ///
995    /// * `request` - The historical data request (can exceed API limits)
996    /// * `continue_on_error` - Whether to continue if a chunk fails (default: false)
997    ///
998    /// # Returns
999    ///
1000    /// A `Result<HistoricalData>` containing all candles from the entire date range
1001    ///
1002    /// # Example
1003    ///
1004    /// ```rust,no_run
1005    /// use kiteconnect_async_wasm::connect::KiteConnect;
1006    /// use kiteconnect_async_wasm::models::market_data::HistoricalDataRequest;
1007    /// use kiteconnect_async_wasm::models::common::Interval;
1008    /// use chrono::NaiveDateTime;
1009    ///
1010    /// # #[tokio::main]
1011    /// # async fn main() -> Result<(), Box<dyn std::error::Error>> {
1012    /// let client = KiteConnect::new("api_key", "access_token");
1013    ///
1014    /// // Request 6 months of 5-minute data (exceeds 90-day limit)
1015    /// let request = HistoricalDataRequest::new(
1016    ///     738561,
1017    ///     NaiveDateTime::parse_from_str("2023-01-01 09:15:00", "%Y-%m-%d %H:%M:%S")?,
1018    ///     NaiveDateTime::parse_from_str("2023-07-01 15:30:00", "%Y-%m-%d %H:%M:%S")?,
1019    ///     Interval::FiveMinute,
1020    /// );
1021    ///
1022    /// // This will automatically split into multiple API calls and combine results
1023    /// let all_data = client.historical_data_chunked(request, false).await?;
1024    /// println!("Retrieved {} candles across the entire 6-month period", all_data.candles.len());
1025    /// # Ok(())
1026    /// # }
1027    /// ```
1028    ///
1029    /// # Performance Considerations
1030    ///
1031    /// - **Optimized for New Instruments**: 60-90% fewer API calls for recently listed stocks
1032    /// - **Early Termination**: Stops immediately when data availability limit is reached
1033    /// - **Rate Limiting**: Automatic delays between chunks (100ms default)
1034    /// - **Memory Usage**: Scales with total number of candles requested
1035    /// - **Deduplication**: Eliminates duplicate data points from overlapping chunks
1036    ///
1037    /// # Error Handling
1038    ///
1039    /// - If `continue_on_error` is `false` (default), the method stops on first error
1040    /// - If `continue_on_error` is `true`, it continues with remaining chunks and logs errors
1041    /// - Successful chunks are still returned even if some chunks fail
1042    /// - Empty chunks trigger early termination (configurable behavior)
1043    pub async fn historical_data_chunked(
1044        &self,
1045        request: HistoricalDataRequest,
1046        continue_on_error: bool,
1047    ) -> KiteResult<HistoricalData> {
1048        // Split the request into valid chunks in reverse chronological order
1049        let chunk_requests = request.split_into_valid_requests_reverse();
1050        
1051        if chunk_requests.len() == 1 {
1052            // No chunking needed, use regular method
1053            return self.historical_data_typed(request).await;
1054        }
1055
1056        #[cfg(feature = "debug")]
1057        log::info!(
1058            "Splitting large historical data request into {} chunks for {} interval (original span: {} days) - processing newest ‚Üí oldest",
1059            chunk_requests.len(),
1060            request.interval,
1061            request.days_span()
1062        );
1063
1064        let mut all_candles = Vec::new();
1065        let mut _successful_chunks = 0;
1066        let mut failed_chunks = 0;
1067
1068        // Process each chunk in reverse chronological order (newest first)
1069        for (i, chunk_request) in chunk_requests.iter().enumerate() {
1070            #[cfg(feature = "debug")]
1071            log::debug!(
1072                "Processing chunk {}/{}: {} to {} ({} days)",
1073                i + 1,
1074                chunk_requests.len(),
1075                chunk_request.from.format("%Y-%m-%d %H:%M:%S"),
1076                chunk_request.to.format("%Y-%m-%d %H:%M:%S"),
1077                chunk_request.days_span()
1078            );
1079
1080            match self.historical_data_typed(chunk_request.clone()).await {
1081                Ok(chunk_data) => {
1082                    if chunk_data.candles.is_empty() {
1083                        #[cfg(feature = "debug")]
1084                        log::info!(
1085                            "Empty chunk encountered at {} ‚Üí {} - reached data availability limit, stopping early",
1086                            chunk_request.from.format("%Y-%m-%d"),
1087                            chunk_request.to.format("%Y-%m-%d")
1088                        );
1089                        
1090                        // Early termination: empty chunk means no more historical data exists
1091                        break;
1092                    }
1093
1094                    #[cfg(feature = "debug")]
1095                    let candles_count = chunk_data.candles.len();
1096                    all_candles.extend(chunk_data.candles);
1097                    _successful_chunks += 1;
1098                    
1099                    #[cfg(feature = "debug")]
1100                    log::debug!(
1101                        "Chunk {}/{} completed successfully: {} candles retrieved",
1102                        i + 1,
1103                        chunk_requests.len(),
1104                        candles_count
1105                    );
1106                }
1107                Err(e) => {
1108                    failed_chunks += 1;
1109                    
1110                    #[cfg(feature = "debug")]
1111                    log::warn!(
1112                        "Chunk {}/{} failed: {:?}",
1113                        i + 1,
1114                        chunk_requests.len(),
1115                        e
1116                    );
1117
1118                    if !continue_on_error {
1119                        return Err(e);
1120                    }
1121                }
1122            }
1123
1124            // Add a small delay between chunks to be respectful to the API
1125            if i < chunk_requests.len() - 1 {
1126                tokio::time::sleep(std::time::Duration::from_millis(100)).await;
1127            }
1128        }
1129
1130        if all_candles.is_empty() && failed_chunks > 0 {
1131            return Err(crate::models::common::KiteError::general(format!(
1132                "All {} chunks failed to retrieve data",
1133                failed_chunks
1134            )));
1135        }
1136
1137        // Sort candles by date to ensure chronological order (oldest ‚Üí newest)
1138        all_candles.sort_by(|a, b| a.date.cmp(&b.date));
1139
1140        #[cfg(feature = "debug")]
1141        log::info!(
1142            "Historical data chunking completed: {} successful chunks, {} failed chunks, {} total candles (processed {} of {} possible chunks)",
1143            _successful_chunks,
1144            failed_chunks,
1145            all_candles.len(),
1146            _successful_chunks + failed_chunks,
1147            chunk_requests.len()
1148        );
1149
1150        // Create the final response
1151        let metadata = HistoricalMetadata {
1152            instrument_token: request.instrument_token,
1153            symbol: format!("Token-{}", request.instrument_token), // We don't have the symbol from chunks
1154            interval: request.interval,
1155            count: all_candles.len(),
1156        };
1157
1158        Ok(HistoricalData {
1159            candles: all_candles,
1160            metadata,
1161        })
1162    }
