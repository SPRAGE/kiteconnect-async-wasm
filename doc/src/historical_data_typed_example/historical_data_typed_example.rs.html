8use chrono::NaiveDateTime;
9use kiteconnect_async_wasm::connect::KiteConnect;
10use kiteconnect_async_wasm::models::common::Interval;
11use kiteconnect_async_wasm::models::market_data::HistoricalDataRequest;
12use std::env;

14/// Get API credentials from environment variables
15fn get_credentials() -> Result<(String, String), Box<dyn std::error::Error>> {
16    let api_key =
17        env::var("KITE_API_KEY").map_err(|_| "KITE_API_KEY environment variable not set")?;
18    let access_token = env::var("KITE_ACCESS_TOKEN")
19        .map_err(|_| "KITE_ACCESS_TOKEN environment variable not set")?;
20    Ok((api_key, access_token))
21}
